# {'<Polygon />'}
 
React representative of `new ol.Polygon(...)` styled.
https://openlayers.org/en/latest/apidoc/module-ol_geom_Polygon-Polygon.html
<Story of={PolygonStories.Primary} />

### Usage
```typescript
import { OSM } from 'ol/source';
import { useRef } from 'react';
import { fromLonLat } from 'ol/proj';
import type { Meta, StoryObj } from '@storybook/react';
import { Map, TileLayer, View, Polygon } from 'react-openlayers';

export default function (props) {
  const mapRef = useRef();

  const coordinates = [[
    [-122.4752749, 37.7500271],
    [-122.4055804, 37.7363170],
    [-122.3909033, 37.7881577],
    [-122.4775065, 37.7882933]
  ]];
  const color = 'rgba(210, 12, 12, 0.5)';
  const center = fromLonLat([-122.42, 37.779]);

  return (
    <Map ref={mapRef}>
      <TileLayer source={new OSM()} />
      <View center={center} zoom={12}/>
      <Polygon coordinates={coordinates} color={color} />
    </Map>
  );
}
```


### props

| Prop        | type    | description |
| ----------- | ------- | ------------|
| coordinates | array   | longitude and latitude that marker to point, e.g. `[[ [-122, 37], [-123, 38], [-124, 39] ]]` | 
| layout      | GeometryLayout | Layout.  | 
| ends        | Arrray  | Ends (for internal use with flat coordinates).  | 
| style       | object  | style  e.g. `{ fill: new Fill({color: 'rgba(210, 12, 12, 0.5)'}) }` |
